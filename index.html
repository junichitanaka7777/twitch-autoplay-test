function loadStream() { if (player) return; // Function to initialization player const initPlayer = () => { embedContainer.style.display = 'block'; player = new Twitch.Player("twitch-embed", { channel: "marbleplanetstreamer", muted: false, parent: ["twitch-autoplay-test.vercel.app"] }); playButton.innerText = "Close Stream"; }; // Check if script is already loaded if (typeof Twitch !== 'undefined') { initPlayer(); } else { // Load script dynamically var script = document.createElement('script'); script.src = 'https://player.twitch.tv/js/embed/v1.js'; script.onload = initPlayer; document.body.appendChild(script); } } function stopStream() { embedContainer.innerHTML = ''; // Remove the player elements embedContainer.style.display = 'none'; player = null; // Reset player instance playButton.innerText = "Watch Stream"; } playButton.addEventListener('click', function () { if (!player) { loadStream(); } else { stopStream(); } }); // Auto-load after 5 seconds setTimeout(loadStream, 5000);
